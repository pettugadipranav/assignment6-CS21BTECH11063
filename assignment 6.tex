\section{Question}
\begin{frame}{Question}
    \begin{block}{\textbf{Papoullis EX 10-7:} } 
    The process $s(t)$ is shot noise with $\lambda =3$ where $h(t)=2$ for $0 \le t \le 10$ and $ h(t)=0$ otherwise. Find $ E\{s(t)\},E\{s^2(t)\},P\{s(7)=0\}$ 
                  \end{block}
     
\end{frame}



\section{Solution}
\begin{frame}{Solution}
\begin{block}{}
Given $\lambda=3$ ,\\
$h(t)=2$     ($ 0\le t\le 10$),\\
$h(t)=0$ otherwise.\\
\end{block}
\begin{block}{}
\eta _s=E\{s(t)\}=\lambda\(\int_{0}^{10} h(t) \,dt\)= \lambda\(\int_{0}^{10} 2 \,dt\)=3\times 2(10-0)=3\times20=60

\sigma_s^2=var\{s(t)\}=\lambda\(\int_{0}^{10} h^2(t) \,dt\)=\lambda\(\int_{0}^{10} 4 \,dt\)=3\times4(10-0)=120

\begin{align}
E\{s^2(t)\} - E\{s(t)\}^2=var\{s(t)\}\\
E\{s^2(t)\} &= E\{s(t)\}^2+var\{s(t)\}\\
&=
3600+120\\
E\{s^2(t)\}=3720
\end{align}
\end{block}
        \end{frame}
        \begin{frame}{}
        s(7)=0 if there are no points in the interval(7-10,7).the number of points in this interval is a poission RV with parameter 10\lambda=30.\\
        
    Hence P\{s(7)=0\}=e^-^3^0
        
        \end{frame}
